Quick Example

example.lucia
fun fibonacci(n: int) -> int:
    if (n <= 1):
        return n
    end
    return fibonacci(n - 1) + fibonacci(n - 2)
end

// Print first 10 Fibonacci numbers
for (i  [1...10]):
    print(fibonacci(i))
end
